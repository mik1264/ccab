    grudger: (history) => history.some(h => h.opponent === 'D') ? 'D' : 'C',
    random: () => Math.random() < 0.5 ? 'C' : 'D',

// Payoff matrix (years in prison)
const payoffs = {
    'CC': [3, 3],  // Both cooperate

    'DD': [1, 1]   // Both defect
};

### Small World Network (Network Science #1)

```javascript
// Watts-Strogatz model
function createSmallWorld(n, k, p) {
    // Start with ring lattice: each node connected to k nearest neighbors
    const edges = createRingLattice(n, k);

    // Rewire each edge with probability p
    for (let edge of edges) {
        if (Math.random() < p) {
            rewireEdge(edge, n);
        }
    }

    // Calculate clustering coefficient and path length
    return {
        edges,
        clustering: calculateClustering(edges, n),
        pathLength: averagePathLength(edges, n)
    };
}
```
